cmake_minimum_required(VERSION 3.10)
project(eagleemu C)

set(CMAKE_C_STANDARD 99)
set(CMAKE_C_STANDARD_REQUIRED ON)

# Emscripten toolchain settings
if(NOT EMSCRIPTEN)
    message(FATAL_ERROR "This CMakeLists is for Emscripten builds only. "
                        "Use: emcmake cmake -S . -B build_emscripten -DCMAKE_TOOLCHAIN_FILE=${EMSCRIPTEN_ROOT}/cmake/Modules/Platform/Emscripten.cmake")
endif()

# SAASound source directory (user must provide SAASound C source)
set(SAASOUND_DIR "${CMAKE_SOURCE_DIR}/../saasound" CACHE PATH "Path to SAASound source directory")
if(NOT EXISTS "${SAASOUND_DIR}/SAASound.c")
    message(WARNING "SAASound source not found at ${SAASOUND_DIR}")
    message(WARNING "Clone it from: https://github.com/kode54/SAASound")
    message(WARNING "  cd ${CMAKE_SOURCE_DIR}/.. && git clone https://github.com/kode54/SAASound saasound")
endif()

# SAASound source files
set(SAASOUND_SOURCES
    ${SAASOUND_DIR}/SAASound.c
    ${SAASOUND_DIR}/SAAImpl.c
    ${SAASOUND_DIR}/SAAFreq.c
    ${SAASOUND_DIR}/SAANoise.c
    ${SAASOUND_DIR}/SAAEnv.c
)

# Emulator source files
set(SOURCES
    8255.c
    cpu.c
    gfx.c
    keyb.c
    main.c
    saa1099_ext.c
    sd.c
    serial.c
    ${SAASOUND_SOURCES}
)

add_executable(eagleemu ${SOURCES})

# Include directories
target_include_directories(eagleemu PRIVATE
    ${SAASOUND_DIR}
)

# Emscripten compile/link flags
set(EMSCRIPTEN_FLAGS
    -sUSE_SDL=2
    -sUSE_SDL_TTF=2
    -sUSE_PTHREADS=1
    -sPTHREAD_POOL_SIZE=2
    -sALLOW_MEMORY_GROWTH=1
    -sINITIAL_MEMORY=268435456
    -sEXIT_RUNTIME=0
    -sEXPORTED_FUNCTIONS="['_main']"
    -sEXPORTED_RUNTIME_METHODS="['ccall','cwrap']"
    -sENVIRONMENT=web,worker
    --preload-file "${CMAKE_SOURCE_DIR}/rom.bin"
    --preload-file "${CMAKE_SOURCE_DIR}/default.ttf"
    --shell-file "${CMAKE_SOURCE_DIR}/eagleemu_shell.html"
    -sFORCE_FILESYSTEM=1
)

target_compile_options(eagleemu PRIVATE ${EMSCRIPTEN_FLAGS})
target_link_options(eagleemu PRIVATE ${EMSCRIPTEN_FLAGS})

# Suppress some SAASound warnings
target_compile_options(eagleemu PRIVATE -Wno-unused-value -Wno-unused-but-set-variable)

set(CMAKE_RUNTIME_OUTPUT_DIRECTORY "${CMAKE_BINARY_DIR}")
